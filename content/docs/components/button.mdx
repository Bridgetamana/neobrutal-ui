## Installation

```bash
npx neobrutal add button
```

## Usage

```tsx
import { Button } from "@/components/ui/button"
```

```tsx
<Button variant="neutral">Button</Button>
```

### Link

Use the `asChild` prop to render as a link.

```tsx
import Link from "next/link"

<Button asChild>
  <Link href="/login">Login</Link>
</Button>
```
