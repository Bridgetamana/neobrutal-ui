## API Reference

### DatePicker

<PropsTable>
  <PropsRow name="value" type="Date | null" defaultValue="null">
    The currently selected date.
  </PropsRow>
  <PropsRow name="onChange" type="(date: Date | null) => void" defaultValue="-">
    Callback fired when the user applies a date selection.
  </PropsRow>
  <PropsRow name="placeholder" type="string" defaultValue="&quot;Select date&quot;">
    Placeholder text shown when no date is selected.
  </PropsRow>
  <PropsRow name="disabled" type="boolean" defaultValue="false">
    Whether the date picker trigger is disabled.
  </PropsRow>
</PropsTable>
