## API Reference

### Modal

<PropsTable>
  <PropsRow name="variant" type="&quot;success&quot; | &quot;warning&quot; | &quot;destructive&quot;" defaultValue="&quot;success&quot;">
    Controls the icon, color scheme, and confirm button style.
  </PropsRow>
  <PropsRow name="open" type="boolean" defaultValue="-">
    Whether the modal is open.
  </PropsRow>
  <PropsRow name="onOpenChange" type="(open: boolean) => void" defaultValue="-">
    Callback fired when the modal open state changes.
  </PropsRow>
  <PropsRow name="title" type="string" defaultValue="-">
    The modal heading text.
  </PropsRow>
  <PropsRow name="description" type="string" defaultValue="-">
    Optional description text below the title.
  </PropsRow>
  <PropsRow name="confirmText" type="string" defaultValue="&quot;Confirm&quot;">
    Label for the confirm action button.
  </PropsRow>
  <PropsRow name="cancelText" type="string" defaultValue="&quot;Cancel&quot;">
    Label for the cancel button.
  </PropsRow>
  <PropsRow name="onConfirm" type="() => void" defaultValue="-">
    Callback fired when the confirm button is clicked.
  </PropsRow>
  <PropsRow name="onCancel" type="() => void" defaultValue="-">
    Callback fired when the cancel button is clicked.
  </PropsRow>
</PropsTable>
