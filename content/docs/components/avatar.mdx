## Installation

```bash
npx neobrutal add avatar
```

## Usage

```tsx
import { Avatar, AvatarImage, AvatarFallback } from "@/components/ui/avatar"
```

```tsx
<Avatar>
  <AvatarImage src="https://github.com/bridgetamana.png" alt="@bridgetamana" />
  <AvatarFallback>CN</AvatarFallback>
</Avatar>
```
