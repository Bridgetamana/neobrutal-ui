## Installation

```bash
npx neobrutal add label
```

## Usage

```tsx
import { Label } from "@/components/ui/label"
```

```tsx
<div className="flex items-center gap-2">
  <Checkbox id="terms" />
  <Label htmlFor="terms">Accept terms and conditions</Label>
</div>
```
