## Installation

```bash
npx neobrutal add alert
```

## Usage

```tsx
import { Alert, AlertTitle, AlertDescription } from "@/components/ui/alert"
```

```tsx
<Alert>
  <AlertTitle>Title</AlertTitle>
  <AlertDescription>Description</AlertDescription>
</Alert>
```
